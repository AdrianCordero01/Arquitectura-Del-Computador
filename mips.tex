\section*{Operaciones de carga y almacenamiento}
\begin{itemize}
  \item \texttt{lw \$t1, 0(\$s0)} \quad Cargar palabra (4 bytes) desde memoria.
  \item \texttt{sw \$t1, 0(\$sp)} \quad Guardar palabra en memoria.
  \item \texttt{lb / lbu} \quad Cargar byte (con o sin signo).
  \item \texttt{la \$a0, etiqueta} \quad Cargar dirección de una etiqueta.
\end{itemize}

\section*{Operaciones aritméticas}
\begin{itemize}
  \item \texttt{add \$t0, \$a1, \$t0} \quad Suma.
  \item \texttt{sub \$t0, \$a2, \$a1} \quad Resta.
  \item \texttt{mul \$v0, \$a0, \$v0} \quad Multiplicación.
  \item \texttt{sll \$s0, \$t0, 2} \quad Shift left lógico (multiplica por 4).
  \item \texttt{sra \$t0, \$t0, 1} \quad Shift right aritmético (divide entre 2).
  \item \texttt{addi \$sp, \$sp, -8} \quad Suma inmediata (ajuste de stack).
\end{itemize}

\section*{Saltos y control de flujo}
\begin{itemize}
  \item \texttt{beq \$t1, \$a3, etiqueta} \quad Branch if equal.
  \item \texttt{bgt \$t1, \$a3, etiqueta} \quad Branch if greater than.
  \item \texttt{ble \$a1, \$a2, etiqueta} \quad Branch if less or equal.
  \item \texttt{j etiqueta} \quad Salto incondicional.
  \item \texttt{jal subrutina} \quad Llamada a subrutina (guarda \$ra).
  \item \texttt{jr \$ra} \quad Retorno de subrutina.
\end{itemize}

\section*{Syscalls}
\begin{itemize}
  \item \texttt{li \$v0, 1} \quad Imprimir entero (\$a0 contiene el valor).
  \item \texttt{li \$v0, 11} \quad Imprimir carácter (\$a0 contiene el código ASCII).
  \item \texttt{li \$v0, 10} \quad Salir del programa.
\end{itemize}

\section*{Ejemplo: Factorial Recursivo}
\begin{lstlisting}
factorial:
    addi $sp, $sp, -8
    sw $a0, 0($sp)
    sw $ra, 4($sp)

    beq $a0, $zero, base_case

    addi $a0, $a0, -1
    jal factorial

    lw $a0, 0($sp)
    mul $v0, $a0, $v0
    j end_factorial

base_case:
    li $v0, 1

end_factorial:
    lw $ra, 4($sp)
    addi $sp, $sp, 8
    jr $ra
\end{lstlisting}
